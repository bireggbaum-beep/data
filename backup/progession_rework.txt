SCHRITT 2: Neue Funktionen hinzufügen (nach getUnlockedCRs())
javascript


function getMaxUnlockedUnit() {
    let maxUnit = UNIT_ORDER[0];
    for (let i = 1; i < UNIT_ORDER.length; i++) {
        const unit = UNIT_ORDER[i];
        const rule = UNIT_UNLOCK[unit];
        if (!rule.from) { maxUnit = unit; continue; }
        const completed = TEXTS
            .filter(t => t.unit === rule.from)
            .filter(t => gameState.completedTexts.includes(t.id))
            .length;
        if (completed >= rule.required) {
            maxUnit = unit;
        } else {
            break;
        }
    }
    return maxUnit;
}

function getUnitCompletionInfo() {
    const maxUnit = getMaxUnlockedUnit();
    const maxIdx = UNIT_ORDER.indexOf(maxUnit);
    const nextUnit = maxIdx < UNIT_ORDER.length - 1 ? UNIT_ORDER[maxIdx + 1] : null;
    
    if (!nextUnit) return { current: maxUnit, label: UNIT_LABELS[maxUnit], next: null, completed: 0, required: 0 };
    
    const rule = UNIT_UNLOCK[nextUnit];
    const completed = TEXTS
        .filter(t => t.unit === rule.from)
        .filter(t => gameState.completedTexts.includes(t.id))
        .length;
    const total = TEXTS.filter(t => t.unit === rule.from).length;
    
    return {
        current: maxUnit,
        label: UNIT_LABELS[maxUnit],
        next: nextUnit,
        nextLabel: UNIT_LABELS[nextUnit],
        completed: completed,
        required: rule.required,
        total: total
    };
}

function isTextUnlocked(text) {
    const crUnlocked = getUnlockedCRs().includes(text.cr);
    const textUnitIdx = UNIT_ORDER.indexOf(text.unit);
    const maxUnitIdx = UNIT_ORDER.indexOf(getMaxUnlockedUnit());
    const unitUnlocked = textUnitIdx !== -1 && textUnitIdx <= maxUnitIdx;
    return crUnlocked && unitUnlocked;
}
